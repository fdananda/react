React Component Composability

O React permite nos referir a componentes dentro de outros componentes.





Exemplo adaptado do capítulo 5 do livro Learning React, de Kirupa Chinnathambi: 

    <script type="text/babel">
        class Quadrado extends React.Component{
            render(){

                const estiloQuadrado = {
                    height: 150,
                    backgroundColor: this.props.color,
                }

                return(
                    <div style={estiloQuadrado}></div>
                );
            }
        }

        class Etiqueta extends React.Component{
            render(){

                const estiloEtiqueta = {
                    fontFamily: "sans-serif",
                    fontWeight: "bold",
                    padding: 13,
                    margin: 0
                }

                return(
                    <p style={estiloEtiqueta}>{this.props.color}</p>
                );
            }
        }

        class Cartao extends React.Component{
            render(){

                const estiloCartao = {
                    height: 200,
                    width: 150,
                    padding: 0,
                    backgroundColor: "#FFF",
                    boxShadow: "0px 0px 5px #666"
                }

                return(
                    <div style={estiloCartao}>
                        <Quadrado color={this.props.color}/>
                        <Etiqueta color={this.props.color}/>
                    </div>
                );
            }
        }

        var cor1 = document.getElementById("corFF6663")
        var cor = cor1.value
        var cor2 = document.getElementById("corFFA737")
        var cor3 = document.getElementById("corFFF")
        var cor4 = document.getElementById("corCCC")

        handleChange()

        cor1.onclick = (e) => {
            if(cor1.checked){
                cor = cor1.value
                handleChange()
            }
        }

        cor2.onclick = (e) => {
            if(cor2.checked){
                cor = cor2.value
                handleChange()
            }
        }

        cor3.onclick = (e) => {
            if(cor3.checked){
                cor = cor3.value
                handleChange()
            }
        }

        cor4.onclick = (e) => {
            if(cor4.checked){
                cor = cor4.value
                handleChange()
            }
        }

        function handleChange() {
         const local = document.getElementById("exemplo")
         
         ReactDOM.render(
            <div>
                <Cartao color={cor} />
            </div>, local
            )
        }        
    </script>
