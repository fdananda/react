React props.children

props.children são usados para adicionar dados entre as tags JSX de abertura e fechamento .

Exemplos: 

    <script type="text/babel">
        var local1 = document.getElementById("exemplo1")
        var local2 = document.getElementById("exemplo2")

        //Class Component
        class Pessoa1 extends React.Component{
            render(){
                return <p>Nome: {this.props.children} </p>
            }
        }

        //Function Component
        function Pessoa2(props){
                return <p>Nome: {props.children} </p>
        }

        ReactDOM.render(<Pessoa1>Fernanda</Pessoa1>, local1)
        ReactDOM.render(<Pessoa2>Helena</Pessoa2>, local2)
    </script>
