React State

O objeto state é onde os valores de propriedade que pertencem ao componente são armazendados. Quando o objeto muda, o componente é renderizado novamente. O state é inicializado no construtor e para alterar o valor, usa-se o método this.setState().

    <script type="text/babel">
                class Pessoa extends React.Component{

            constructor(props){
                super(props)
                this.state = {
                    nome: "Fernanda",
                    sobrenome: "Moreira",
                    ano: 1977
                }
            }

            mudarPessoa = () => {
                this.setState({nome: "Helena", sobrenome: "Rodrigues", ano: 1987})
            }

            render(){
                return(
                    <>
                     <p>Nome: {this.state.nome}</p>
                     <p>Sobrenome: {this.state.sobrenome}</p>
                     <p>Ano: {this.state.ano}</p>

                     <button onClick={this.mudarPessoa}>Mudar Pessoa</button>
                    </>

                );
            }
        }

         const local = document.getElementById("exemplo")
         
         ReactDOM.render(
            <div>
                <Pessoa />
            </div>, local)
    </script>
