React Spread Operator

O Spread Operator ( ...) permite copiar rapidamente todo ou parte de um array ou objeto existente para outro array ou objeto.

Exemplo adaptado do capítulo 6 do livro Learning React, de Kirupa Chinnathambi: 

    <script type="text/babel">
                class TextoExibicao extends React.Component{
            render(){
                return(
                    <div>
                        <strong>Nome:</strong> {this.props.nome}<br/>
                        <strong>Sobrenome:</strong> {this.props.sobrenome}<br/>
                        <strong>Idade:</strong> {this.props.idade}<br/>
                    </div>
                );
            }
        }

        class Etiqueta extends React.Component{
            render(){
                return(
                    <<TextoExibicao {...this.props} />
                );
            }
        }

        class Pessoa extends React.Component{
            render(){
                return(
                    <div>
                     <Etiqueta {...this.props} />
                    </div>
                );
            }
        }

         const local = document.getElementById("exemplo")
         
         ReactDOM.render(
            <div>
                <Pessoa nome="Fernanda" sobrenome="Moreira" idade="47" />
            </div>, local
            )
                
    </script>
