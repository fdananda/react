React Props

Props são argumentos passados para componentes React.

Exemplos: 

    <script type="text/babel">
        var local1 = document.getElementById("exemplo1")
        var local2 = document.getElementById("exemplo2")

        //Class Component
        class Pessoa1 extends React.Component{
            render(){
                return <p>Nome: {this.props.nome} </p>
            }
        }

        //Function Component
        function Pessoa2(props){
                return <p>Nome: {props.nome} </p>
        }

        ReactDOM.render(<Pessoa1 nome="Fernanda"/>, local1)
        ReactDOM.render(<Pessoa2 nome="Helena"/>, local2)
    </script>
