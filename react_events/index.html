React Events

o React pode executar ações com base nos eventos do usuário. Os eventos React são escritos na sintaxe camelCase: onClick em vez de onclick. Os manipuladores de eventos são escritos entre chaves: onClick={shoot} em vez de onClick="shoot()". Para passar um argumento para um manipulador de eventos, usa-se uma função de seta. 

    <script type="text/babel">
        class Pessoa extends React.Component{

            constructor(props){
                super(props)
                this.state = {
                    nome: "Fernanda",
                    sobrenome: "Moreira"
                }
            }

            mudarPessoa = () => {
            this.setState({nome: "Helena"})
         }

            render(){
                return(
                    <>
                     <p>Nome: {this.state.nome}</p>
                     <p>Sobrenome: {this.state.sobrenome}</p>
                    <button onClick={this.mudarPessoa}>Mudar Pessoa</button>
                    </>

                );
            }
        }        

         const local = document.getElementById("exemplo")
         
         ReactDOM.render(
            <div>
                <Pessoa />
            </div>, local)   
    </script>
