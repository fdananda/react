React Fragment

O React Fragment, que também pode ser utilizado via <>...</>, permite agrupar elementos sem um nó. Opcionalmente, pode-se utilizar "key" com o Fragment explicitando os dados.

    <script type="text/babel">
          class TextoExibicao extends React.Component{
            render(){
                return(
                    <React.Fragment>
                        <strong>Nome:</strong> {this.props.nome}<br/>
                        <strong>Sobrenome:</strong> {this.props.sobrenome}<br/>
                        <strong>Idade:</strong> {this.props.idade}<p/>

                        <>
                        <strong>Nome:</strong> {this.props.nome}<br/>
                        <strong>Sobrenome:</strong> {this.props.sobrenome}<br/>
                        <strong>Idade:</strong> {this.props.idade}<br/>
                        </>

                        <p key={this.props.nome}><strong>Nome:</strong> {this.props.nome}</p>
                        <p key={this.props.sobrenome}><strong>Sobrenome:</strong> {this.props.sobrenome}</p>
                        <p key={this.props.idade}><strong>Idade:</strong> {this.props.idade}</p>

                    </React.Fragment>
                );
            }
        }

        class Pessoa extends React.Component{
            render(){
                return(
                    <div>
                     <TextoExibicao {...this.props} />
                    </div>
                );
            }
        }

         const local = document.getElementById("exemplo")
         
         ReactDOM.render(
            <div>
                <Pessoa nome="Fernanda" sobrenome="Moreira" idade="45" />
            </div>, local
            )  
    </script>
